REPORT ZEDR13_03.

DATA : gs_data LIKE sflight,
      gt_data LIKE TABLE OF sflight.

CLEAR : gt_data,gs_data.

SELECT CARRID CONNID FLDATE PRICE CURRENCY PLANETYPE
  INTO CORRESPONDING FIELDS OF TABLE gt_data
  FROM sflight
  WHERE CARRID = 'AZ'.



LOOP AT gt_data INTO gs_data.
WRITE :
gs_data-CARRID,
gs_data-CONNID,
gs_data-FLDATE,
gs_data-PRICE,
gs_data-CURRENCY,
gs_data-PLANETYPE,/.
ENDLOOP.
