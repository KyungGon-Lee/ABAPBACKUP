FORM get_data .
    CLEAR : gt_data, gt_data[].

  SELECT BUKRS WERKS MATNR MENGE MEINS
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM ZTUI513_02
    WHERE MATNR IN S_MATNR.


ENDFORM.

FORM save_data.
  CALL METHOD go_grid->check_changed_data.

  DATA : lt_sav TYPE TABLE OF ZTUI513_02 WITH HEADER LINE.

  CLEAR : lt_sav, lt_sav[], gs_row, gt_row.

  CALL METHOD go_grid->get_selected_rows
    IMPORTING
      et_index_rows = gt_row.

  IF gt_row IS INITIAL.
    MESSAGE s000 WITH 'Please Select row' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  LOOP AT gt_row INTO gs_row.
    READ TABLE gt_data INDEX gs_row-index.

    IF sy-subrc EQ 0.

      MOVE-CORRESPONDING gt_data TO lt_sav.
      IF sy-subrc EQ 0.
        APPEND lt_sav.
        CLEAR lt_sav.

      ENDIF.
    ENDIF.
  ENDLOOP.

  IF lt_sav[] IS NOT INITIAL.
    MODIFY ZTUI513_02 FROM TABLE lt_sav.
    IF sy-subrc EQ 0.

      COMMIT WORK AND WAIT.
      MESSAGE s000 WITH 'Save Success'.
    ELSE.
      ROLLBACK WORK.
      MESSAGE s000 WITH 'Save fail' DISPLAY LIKE 'E'.

    ENDIF.

  ENDIF.
ENDFORM.

FORM delete_data .
 DATA : lt_del TYPE TABLE OF ZTUI513_02 WITH HEADER LINE.

  CLEAR : lt_del, lt_del[], gs_row, gt_row.
  CALL METHOD go_grid->get_selected_rows
    IMPORTING
      et_index_rows = gt_row.

  IF gt_row IS INITIAL.
    MESSAGE s000 WITH 'Please Select row' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  LOOP AT gt_row INTO gs_row.
    READ TABLE gt_data INDEX gs_row-index.

    IF sy-subrc EQ 0.
      MOVE-CORRESPONDING gt_data TO lt_del.


      APPEND lt_del.
      CLEAR lt_del.
    ENDIF.
  ENDLOOP.

  IF lt_del[] IS NOT INITIAL.
    DELETE ZTUI513_02 FROM TABLE lt_del.

    IF sy-subrc EQ 0.
      COMMIT WORK AND WAIT.

      MESSAGE s000 WITH 'Delete Success.'.
      PERFORM get_data.

    ELSE.

      ROLLBACK WORK.
      MESSAGE s000 WITH 'Delete fail' DISPLAY LIKE 'E'.
    ENDIF.
  ENDIF.

ENDFORM.
